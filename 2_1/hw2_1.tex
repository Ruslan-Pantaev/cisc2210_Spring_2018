$p =$ "it is raining,"\\
$q =$ "the sun is shining,"\\

\hfill \break

\end{center}

\subsection*{(a)}
\begin{center}

\hfill \break
$p \wedge q$

\hfill \break
$\neg r \rightarrow q$

\subsection*{12.}
\begin{center}
Find counterexamples to the following assertions:
\end{center}

\subsection*{(a)}
\begin{center}
$2^{n} - 1$ is prime for every $n \geq 2$\\
\hfill \break
Consider $h(n) = |\{k \in \mathbb{N} : k|n\}|$,\\
if $h(n) = 2$, then $n$ is prime.\\
let $n = 4$,\\
$2^{4} - 1 = 15$,\\
$h(15) = |1,3,5,15| = 4$,\\
$4 \neq 2$,\\
thus $n = 4$ is a counterexample.\\
let $n = 6$,\\
$2^{4} - 1 = 63$,\\
$h(63) = |1,3,7,9,21,63| = 6$,\\
$6 \neq 2$,\\
thus $n = 6$ is another counterexample.
\end{center}
